### Preguntas por mes
```SQL
-- Contar preguntas por mes (Agrupación histórica)
SELECT
    -- Crea una fecha limpia (Día 1 de cada mes) para que sea fácil de graficar
    DATEFROMPARTS(YEAR(CreationDate), MONTH(CreationDate), 1) AS [Mes],
    COUNT(*) AS [Cantidad de Preguntas]
FROM Posts
WHERE
    PostTypeId = 1 -- El 1 filtra solo "Preguntas" (2 sería respuestas)
GROUP BY
    YEAR(CreationDate), MONTH(CreationDate)
ORDER BY
    YEAR(CreationDate), MONTH(CreationDate);
```
### Comentarios por mes
```SQL
-- Contar Comentarios por mes (Tendencia de discusión)
SELECT
    DATEFROMPARTS(YEAR(CreationDate), MONTH(CreationDate), 1) AS [Mes],
    COUNT(*) AS [Total Comentarios]
FROM Comments
GROUP BY
    YEAR(CreationDate), MONTH(CreationDate)
ORDER BY
    YEAR(CreationDate), MONTH(CreationDate);
```
### Preguntas sin respuesta
```SQL
-- Preguntas sin respuesta (AnswerCount = 0)
SELECT
    DATEFROMPARTS(YEAR(CreationDate), MONTH(CreationDate), 1) AS [Mes],
    SUM(CASE WHEN AnswerCount = 0 THEN 1 ELSE 0 END) * 100.0 / COUNT(*) AS [Porcentaje Sin Respuesta]
FROM Posts
WHERE
    PostTypeId = 1 -- Solo preguntas
GROUP BY
    YEAR(CreationDate), MONTH(CreationDate)
ORDER BY
    YEAR(CreationDate), MONTH(CreationDate);
```
### El "Efecto ChatGPT por Lenguaje": Comparativa de Tags
```SQL
-- Cuales son los tags en programacion
SELECT
    DATEFROMPARTS(YEAR(p.CreationDate), MONTH(p.CreationDate), 1) AS [Mes],
    t.TagName,
    COUNT(*) AS [Total Preguntas]
FROM Posts p
JOIN PostTags pt ON p.Id = pt.PostId
JOIN Tags t ON pt.TagId = t.Id
WHERE
    t.TagName IN ('python', 'css','javascript', 'c++', 'rust') -- Tus tags de interés
    AND p.CreationDate >= '2008-01-01' -- Enfocado en la era reciente
GROUP BY
    YEAR(p.CreationDate), MONTH(p.CreationDate), t.TagName
ORDER BY
    YEAR(p.CreationDate), MONTH(p.CreationDate);
```

```SQL
-- Métricas combinadas: Preguntas, Respuestas, Comentarios, Votos
WITH MetricasPosts AS (
    SELECT
        DATEFROMPARTS(YEAR(CreationDate), MONTH(CreationDate), 1) AS [Mes],
        -- Contamos preguntas y respuestas usando el ID de tipo
        SUM(CASE WHEN PostTypeId = 1 THEN 1 ELSE 0 END) AS [Preguntas],
        SUM(CASE WHEN PostTypeId = 2 THEN 1 ELSE 0 END) AS [Respuestas],
        -- Métrica de calidad (solo aplica a preguntas)
        SUM(CASE WHEN PostTypeId = 1 AND AnswerCount = 0 THEN 1 ELSE 0 END) AS [PreguntasSinRespuesta]
    FROM Posts
    WHERE PostTypeId IN (1, 2)
    GROUP BY YEAR(CreationDate), MONTH(CreationDate)
),
MetricasComentarios AS (
    SELECT
        DATEFROMPARTS(YEAR(CreationDate), MONTH(CreationDate), 1) AS [Mes],
        COUNT(*) AS [Comentarios]
    FROM Comments
    GROUP BY YEAR(CreationDate), MONTH(CreationDate)
),
MetricasVotos AS (
    SELECT
        DATEFROMPARTS(YEAR(CreationDate), MONTH(CreationDate), 1) AS [Mes],
        COUNT(*) AS [Votos]
    FROM Votes 
    WHERE VoteTypeId IN (2, 3)
    GROUP BY YEAR(CreationDate), MONTH(CreationDate)
)
SELECT
    p.[Mes],
    p.[Preguntas],
    p.[Respuestas],
    -- Calculamos el ratio de respuestas por pregunta           
    CAST(p.[Respuestas] AS FLOAT) / NULLIF(p.[Preguntas], 0) AS [Ratio Respuestas/Pregunta],
    c.[Comentarios] AS [Total Comentarios],
    v.[Votos] AS [Total Votos],
    (p.[PreguntasSinRespuesta] * 100.0 / NULLIF(p.[Preguntas], 0)) AS [Porcentaje Sin Respuesta]
FROM MetricasPosts p
LEFT JOIN MetricasComentarios c ON p.[Mes] = c.[Mes]
LEFT JOIN MetricasVotos v ON p.[Mes] = v.[Mes]
ORDER BY p.[Mes] DESC;
```

### Evolución de la "Frustración" (Comentarios por Respuesta)
Esta consulta mide cuántos comentarios se necesitan para generar una 
respuesta. 
Un aumento en este ratio sugiere que las preguntas son de menor calidad 
(requieren más aclaraciones) o que la comunidad se ha vuelto más pedante.
```SQL
--- Comentarios por Respuesta a lo largo del tiempo
SELECT 
    DATEADD(month, DATEDIFF(month, 0, CreationDate), 0) AS [Month],
    CAST(SUM(CommentCount) AS FLOAT) / NULLIF(SUM(AnswerCount), 0) AS CommentsPerAnswer,
    AVG(CAST(AnswerCount AS FLOAT)) AS AvgAnswerCount
FROM Posts
WHERE PostTypeId = 1 -- Solo Preguntas
AND CreationDate > '2019-01-01'
GROUP BY DATEADD(month, DATEDIFF(month, 0, CreationDate), 0)
ORDER BY [Month]
```

{/*  QueryResults-PercentNegativeScore */}
### Distribución de Calidad (Score vs. Respuestas)
El Score (votos a favor menos votos en contra) es el indicador definitivo de calidad según la comunidad. Esta consulta agrupa por mes para ver si las preguntas con "puntuación negativa" están aumentando.
```SQL
--- Puntuación promedio y porcentaje de puntuaciones negativas
SELECT 
    DATEADD(month, DATEDIFF(month, 0, CreationDate), 0) AS [Month],
    AVG(CAST(Score AS FLOAT)) AS AvgScore,
    SUM(CASE WHEN Score < 0 THEN 1 ELSE 0 END) * 100.0 / COUNT(*) AS PercentNegativeScore
FROM Posts
WHERE PostTypeId = 1
AND CreationDate > '2019-01-01'
GROUP BY DATEADD(month, DATEDIFF(month, 0, CreationDate), 0)
ORDER BY [Month]
```

### Tasa de Deserción de "Power Users"
Esta consulta es crucial para tu hipótesis sobre los LLM. Analiza a los usuarios que eran el motor del sitio (expertos con >2000 de reputación) y observa cuántos de ellos han dejado de publicar respuestas a partir de la llegada de ChatGPT (finales de 2022).
```SQL
/* Usuarios expertos que estaban activos en 2021-2022 y su actividad actual */
WITH Expertos AS (
    SELECT Id, DisplayName, Reputation
    FROM Users
    WHERE Reputation > 2000
)
SELECT 
    DATEADD(month, DATEDIFF(month, 0, p.CreationDate), 0) AS [Mes],
    COUNT(DISTINCT p.OwnerUserId) AS ExpertosActivos,
    COUNT(p.Id) AS TotalRespuestasExpertos
FROM Posts p
JOIN Expertos e ON p.OwnerUserId = e.Id
WHERE p.PostTypeId = 2 -- Solo respuestas
AND p.CreationDate > '2008-08-01'
GROUP BY DATEADD(month, DATEDIFF(month, 0, p.CreationDate), 0)
ORDER BY [Mes]
```

### Nuevas Cuentas y su "Supervivencia"
Esta consulta mide cuántos usuarios se registran por mes y, lo más importante, cuántos de ellos realmente contribuyen. Si hay muchos registros pero poca actividad, son "cuentas fantasma".
```SQL
SELECT 
    DATEADD(month, DATEDIFF(month, 0, CreationDate), 0) AS [MesRegistro],
    COUNT(*) AS NuevosUsuarios,
    SUM(CASE WHEN UpVotes + DownVotes > 0 THEN 1 ELSE 0 END) AS UsuariosQueVotaron,
    SUM(CASE WHEN Views > 10 THEN 1 ELSE 0 END) AS UsuariosConPerfilVisto
FROM Users
WHERE CreationDate > '2008-08-01'
GROUP BY DATEADD(month, DATEDIFF(month, 0, CreationDate), 0)
ORDER BY [MesRegistro]
```
## MEdir la complejidad de las preguntas

-- ============================================================================
-- SIMPLE: Monthly Average Question and Answer Length (2008 - Present)
-- ============================================================================
-- Shows the average length of questions and answers per month
-- Perfect for creating a time series chart
-- ============================================================================
```SQL
SELECT 
    YEAR(q.CreationDate) AS Year,
    MONTH(q.CreationDate) AS Month,
    DATEFROMPARTS(YEAR(q.CreationDate), MONTH(q.CreationDate), 1) AS Date,
    
    -- Question metrics
    COUNT(q.Id) AS TotalQuestions,
    AVG(LEN(q.Body)) AS AvgQuestionLength,
    
    -- Answer metrics (average of all answers posted that month)
    (SELECT AVG(LEN(a.Body)) 
     FROM Posts a 
     WHERE a.PostTypeId = 2 
       AND YEAR(a.CreationDate) = YEAR(q.CreationDate)
       AND MONTH(a.CreationDate) = MONTH(q.CreationDate)) AS AvgAnswerLength

FROM Posts q

WHERE 
    q.PostTypeId = 1  -- Questions only
    AND q.CreationDate >= '2008-07-31'  -- Stack Overflow launch
    AND q.DeletionDate IS NULL

GROUP BY 
    YEAR(q.CreationDate), 
    MONTH(q.CreationDate)

ORDER BY 
    Year, Month;
```
